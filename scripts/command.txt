# MediaManagement command reference

# 1) Move to project root
cd /Users/zhy/Documents/GitHub/MediaManagement

# 2) Create virtual environment (Python 3)
python3 -m venv .venv

# 3) Activate virtual environment
# macOS/Linux:
source .venv/bin/activate
# Windows (PowerShell):
# .venv\Scripts\Activate.ps1
# Windows (cmd):
# .venv\Scripts\activate.bat

# 3.1) Verify interpreter and pip are from .venv
which python
python -m pip --version

# 4) Upgrade pip tooling (optional but recommended)
python -m pip install --upgrade pip setuptools wheel

# 5) Install project dependencies
python -m pip install -r requirements.txt

# 6) Run the web app in dev mode
python -m uvicorn app:app --reload --host 127.0.0.1 --port 8000

# 7) Open endpoints for quick verification
# App UI: http://127.0.0.1:8000/
# API docs: http://127.0.0.1:8000/docs

# 8) Debug API from terminal (run in another shell while server is up)
curl -s http://127.0.0.1:8000/api/feeds
curl -s http://127.0.0.1:8000/api/settings

# 9) Use built-in debug scripts
python scripts/debug.py
python scripts/debug_rss.py

# 10) Inspect runtime storage and logs
cat storage/storage.json
ls -la storage/logs

# 11) Deactivate virtual environment when done
deactivate
